## hud.zig

新增了计算背景框的位置和大小的方法，然后进行绘制。

```zig
const std = @import("std");
const zhu = @import("zhu");

const batch = zhu.batch;
const text = zhu.text;

const com = @import("component.zig");
const spawn = @import("spawn.zig");

const ImageArea = struct {
    name: []const u8,
    image: [:0]const u8,
    rect: zhu.Rect,
};

const UiZon = struct {
    icon: []const ImageArea,
    border: []const ImageArea,
    face: []const ImageArea,
    layout: struct {
        padding: f32,
        frameSize: zhu.Vector2,
        fontOffset: zhu.Vector2,
    },
};

const SessionData = struct {
    level: u32,
    point: u32,
    units: []const struct {
        face: u32,
        class: com.PlayerEnum,
        level: u32,
        rarity: u32,
    },
};

const uiZon: UiZon = @import("zon/ui.zon");
const ctxZon: SessionData = @import("zon/context.zon");

// --- 肖像槽位 ---
const UnitSlot = struct {
    face: u32,
    class: com.PlayerEnum,
    cost: u8,
    rarity: u8,
    faceImage: zhu.graphics.Image,
    frame: zhu.graphics.Image,
    icon: zhu.graphics.Image,
    rect: zhu.Rect,
};

var slots: [ctxZon.units.len]UnitSlot = undefined;
var backgroundRect: zhu.Rect = undefined;
var hoveredIndex: ?usize = null;

fn findFrame(rarity: u8) ?*const ImageArea {
    const frameName: []const u8 = if (rarity >= 2) "rare" else "common";
    for (uiZon.border) |*f| {
        if (std.mem.eql(u8, f.name, frameName)) return f;
    }
    return null;
}

fn getIcon(class: com.PlayerEnum) ImageArea {
    return uiZon.icon[@intFromEnum(class)];
}

// --- 初始化 ---

pub fn init() void {

    // 计算背景条宽度和起始位置
    computeBackgroundRect(@floatFromInt(ctxZon.units.len));
}

pub fn deinit() void {}

// --- 更新 ---

pub fn update() void {
}

fn computeBackgroundRect(count: f32) void {
    const padding = uiZon.layout.padding;
    const size = uiZon.layout.frameSize;

    // 计算总宽度和起始位置
    const totalWidth = (size.x + padding) * count + padding;
    const startX = (zhu.window.size.x - totalWidth) / 2;
    const startY = zhu.window.size.y - size.y - 2 * padding;

    backgroundRect = .{
        .min = .xy(startX, startY - padding), // 再往上挪一点，给边框留空间
        .size = .xy(totalWidth, size.y + 2 * padding),
    };
}

// --- 渲染 ---

pub fn draw() void {
    // 背景条
    batch.drawRect(backgroundRect, .{ .color = .gray(0.1, 0.1) });
}
```
