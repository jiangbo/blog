## hud.zig

抬头显示加载了这个文件，并且打印出了相关的信息。

```zig
const std = @import("std");
const zhu = @import("zhu");

const session = @import("session.zig");
const spawn = @import("spawn.zig");

const batch = zhu.batch;
const text = zhu.text;
const assets = zhu.assets;
const graphics = zhu.graphics;

const Vector2 = zhu.Vector2;
const Rect = zhu.Rect;
const Color = graphics.Color;
const Image = graphics.Image;

const PlayerEnum = session.PlayerEnum;

// --- UI 配置数据类型 ---
const ImageArea = struct {
    name: []const u8,
    image: [:0]const u8,
    rect: Rect,
};

const UiZon = struct {
    icon: []const ImageArea,
    border: []const ImageArea,
    face: []const ImageArea,
    layout: struct {
        padding: f32,
        fontSize: f32,
        frameSize: Vector2,
        fontOffset: Vector2,
    },
};

const uiZon: UiZon = @import("zon/ui.zon");

// --- 肖像槽位 ---

const PortraitSlot = struct {
    name: []const u8,
    playerEnum: PlayerEnum,
    cost: u8,
    rarity: u8,
    face: Image,
    frame: Image,
    icon: Image,
    rect: Rect,
};

const slotCount = session.getSlots().len;
var slots: [slotCount]PortraitSlot = undefined;
var barRect: Rect = undefined;
var hoveredIndex: ?usize = null;

// --- 初始化 ---
pub fn init() void {
    for (uiZon.face) |face| {
        std.log.info("name: {s}, image: {s}", .{ face.name, face.image });
    }
}

pub fn deinit() void {}

pub fn update() void {}

pub fn draw() void {}

```
